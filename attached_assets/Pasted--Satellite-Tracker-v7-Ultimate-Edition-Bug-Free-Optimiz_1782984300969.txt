float degrees_to_radians(float d) { return d * PI / 180.0f; }

float calc_local_solar_hour(int utc_hour, float lon_deg) {
  float h = (float)utc_hour + (lon_deg / 15.0f);
  while (h <  0.0f)  h += 24.0f;
  while (h >= 24.0f) h -= 24.0f;
  return h;
}

float calculate_mercator_x(float lon_deg, float w) { return (lon_deg + 180.0f) * (w / 360.0f); }
float calculate_mercator_y(float lat_deg, float h, float w) {
  float r = degrees_to_radians(lat_deg);
  float n = log(tan(PI / 4.0f + r / 2.0f));
  return (h / 2.0f) - (w * n / (2.0f * PI));
}

struct GeoCache { float lat; float lon; char name[48]; };
static GeoCache s_geo_cache[8];
static int s_cache_len = 0;
static int s_cache_idx = 0;

String get_country_name_from_coords(float lat, float lon, uint32_t mutex_timeout_ms = 6000) {
  if (isnan(lat) || isnan(lon) || fabsf(lat) > 90.0f || fabsf(lon) > 180.0f
      || (lat == 0.0f && lon == 0.0f)) {
    return "Searching...";
  }
  for (int i = 0; i < s_cache_len; i++) {
    if (fabsf(s_geo_cache[i].lat - lat) < 0.8f &&
        fabsf(s_geo_cache[i].lon - lon) < 0.8f) {
      return String(s_geo_cache[i].name);
    }
  }
  if (WiFi.status() != WL_CONNECTED) {
    return guess_location_name(lat, lon);
  }
  bool api_ran  = false;
  String result = "";
  if (xSemaphoreTake(g_http_mutex, pdMS_TO_TICKS(mutex_timeout_ms)) == pdTRUE) {
    WiFiClient plain_client;
    plain_client.setTimeout(5000);
    HTTPClient http;
    http.setTimeout(5000);
    http.addHeader("User-Agent", "ESP32-SatTracker/1.0");
    String url = "http://api.bigdatacloud.net/data/reverse-geocode-client?latitude="
                 + String(lat, 4) + "&longitude=" + String(lon, 4)
                 + "&localityLanguage=en";
    if (http.begin(plain_client, url)) {
      int code = http.GET();
      if (code == 200) {
        StaticJsonDocument<512> doc;
        if (deserializeJson(doc, http.getStream()) == DeserializationError::Ok) {
          api_ran = true;
          const char* country = doc["countryName"] | "";
          if (strlen(country) > 0 && strcmp(country, "null") != 0) {
            result = String(country);
          } else {
            result = guess_location_name(lat, lon);
          }
        }
      }
      http.end();
    }
    xSemaphoreGive(g_http_mutex);
  }
  if (!api_ran || result.length() == 0) {
    result = guess_location_name(lat, lon);
  }
  s_geo_cache[s_cache_idx].lat = lat;
  s_geo_cache[s_cache_idx].lon = lon;
  strncpy(s_geo_cache[s_cache_idx].name, result.c_str(), 47);
  s_geo_cache[s_cache_idx].name[47] = '\0';
  s_cache_idx = (s_cache_idx + 1) % 8;
  if (s_cache_len < 8) s_cache_len++;
  return result;
}

// =====================================================================
// Fetch Kp index (NOAA)
// =====================================================================
void fetch_kp_index() {
  if (WiFi.status() != WL_CONNECTED) return;
  if (xSemaphoreTake(g_http_mutex, pdMS_TO_TICKS(5000)) == pdTRUE) {
    WiFiClientSecure client;
    client.setInsecure(); client.setTimeout(4000);
    HTTPClient http; http.setTimeout(4000);
    if (http.begin(client, "https://services.swpc.noaa.gov/products/noaa-planetary-k-index.json")) {
      int code = http.GET();
      if (code == HTTP_CODE_OK) {
        String payload = http.getString();
        DynamicJsonDocument doc(4096);
        if (!deserializeJson(doc, payload) && doc.is<JsonArray>() && doc.size() > 1) {
          JsonArray last = doc[doc.size() - 1];
          if (last.size() >= 2) {
            float kp = last[1].as<String>().toFloat();
            g_kp_index = kp;
            if      (kp < 4.0f) strcpy(g_space_weather_status, "QUIET");
            else if (kp < 6.0f) strcpy(g_space_weather_status, "MODERATE");
            else                strcpy(g_space_weather_status, "STORM!");
          }
        }
      }
      http.end();
    }
    xSemaphoreGive(g_http_mutex);
  }
}

void update_ionospheric_delay() {
  DateTime dt(unix_time);
  float solar_hour = device_loc_known ? calc_local_solar_hour(dt.hour(), device_lon) : (float)dt.hour();
  float base_tec = 5.0f;
  if (solar_hour >= 8.0f && solar_hour <= 18.0f)
    base_tec = 25.0f - fabsf(13.0f - solar_hour) * 2.0f;
  g_tec_value = base_tec + (g_kp_index * 1.5f);
  double tec_total    = (double)g_tec_value * 1e16;
  double freq_hz      = (double)ISS_FREQ_MHZ * 1e6;
  double path_delay_m = (40.3 * tec_total) / (freq_hz * freq_hz);
  double c_m_per_s    = (double)SPEED_OF_LIGHT * 1000.0;
  g_iono_delay_ns     = (float)((path_delay_m / c_m_per_s) * 1e9);
}

// =====================================================================
// Antenna Pointing & Telecom Math
// =====================================================================
float calc_central_angle_rad(float obs_lat, float obs_lon, float sat_lat, float sat_lon) {
  float dlat = degrees_to_radians(sat_lat - obs_lat);
  float dlon = degrees_to_radians(sat_lon - obs_lon);
  float a = sinf(dlat/2)*sinf(dlat/2) + cosf(degrees_to_radians(obs_lat))*cosf(degrees_to_radians(sat_lat))*sinf(dlon/2)*sinf(dlon/2);
  return 2.0f * atan2f(sqrtf(a), sqrtf(1.0f - a));
}

float calc_range_km(float obs_lat, float obs_lon, float sat_lat, float sat_lon, float alt_km) {
  float ca = calc_central_angle_rad(obs_lat, obs_lon, sat_lat, sat_lon);
  float Re = EARTH_R_KM, h = alt_km;
  return sqrtf(Re*Re + (Re+h)*(Re+h) - 2.0f*Re*(Re+h)*cosf(ca));
}

float calc_elevation_deg(float obs_lat, float obs_lon, float sat_lat, float sat_lon, float alt_km) {
  float ca = calc_central_angle_rad(obs_lat, obs_lon, sat_lat, sat_lon);
  float Re = EARTH_R_KM, h = alt_km;
  return atan2f(cosf(ca) - Re/(Re+h), sinf(ca)) * 180.0f / PI;
}

float calc_azimuth_deg(float obs_lat, float obs_lon, float sat_lat, float sat_lon) {
  float phi1 = degrees_to_radians(obs_lat);
  float phi2 = degrees_to_radians(sat_lat);
  float dl   = degrees_to_radians(sat_lon - obs_lon);
  float y = sinf(dl) * cosf(phi2);
  float x = cosf(phi1) * sinf(phi2) - sinf(phi1) * cosf(phi2) * cosf(dl);
  float az = atan2f(y, x) * 180.0f / PI;
  return fmodf(az + 360.0f, 360.0f);
}

const char* az_to_cardinal(float az) {
  static const char* const cards[8] = {"N","NE","E","SE","S","SW","W","NW"};
  return cards[(int)((az + 22.5f) / 45.0f) % 8];
}

void update_pass_state(float cur_el, float prev_el, char* buf, size_t sz) {
  const char* s;
  if      (cur_el < 0.0f)            s = "NO SIGNAL";
  else if (cur_el > prev_el + 0.05f) s = "RISING";
  else if (cur_el < prev_el - 0.05f) s = "SETTING";
  else                               s = "PEAK";
  strncpy(buf, s, sz - 1); buf[sz - 1] = '\0';
}

void update_compass_needle_pos(lv_obj_t* ring, lv_obj_t* needle, lv_obj_t* head, lv_point_precise_t* pts, float az, float el, bool is_iss) {
  if (!ring || !needle) return;

  // مركز البوصلة بالضبط (الحلقة مقاسها 80x80 فمركزها 40,40 — ده الطرف الثابت)
  const float CX = 40.0f, CY = 40.0f;
  const float R  = 32.0f;   // طول السهم من المركز لحد قرب حافة الحلقة

  if (el < 0.0f) {
    lv_obj_set_style_border_opa(ring, LV_OPA_30, 0);
    lv_obj_add_flag(needle, LV_OBJ_FLAG_HIDDEN);
    if (head) lv_obj_add_flag(head, LV_OBJ_FLAG_HIDDEN);
  } else {
    lv_obj_set_style_border_opa(ring, LV_OPA_COVER, 0);
    lv_obj_clear_flag(needle, LV_OBJ_FLAG_HIDDEN);

    // نطرح 90 درجة عشان زاوية صفر (شمال) تتجه لفوق على الشاشة
    float screen_rad = degrees_to_radians(az - 90.0f);
    float ex = CX + R * cosf(screen_rad);
    float ey = CY + R * sinf(screen_rad);

    // الطرف الثابت = مركز البوصلة بالضبط (نفس مكان النقطة البيضاء)
    // الطرف المتحرك = اتجاه القمر الصناعي الحالي
    pts[0].x = (lv_value_precise_t)CX;
    pts[0].y = (lv_value_precise_t)CY;
    pts[1].x = (lv_value_precise_t)ex;
    pts[1].y = (lv_value_precise_t)ey;
    lv_line_set_points(needle, pts, 2);

    if (head) {
      lv_obj_clear_flag(head, LV_OBJ_FLAG_HIDDEN);
      lv_obj_set_pos(head, (int)ex - 5, (int)ey - 6);
    }
  }
}

float calc_doppler_hz(float freq_mhz, float range_now_km, float range_prev_km, float dt_sec) {
  if (dt_sec <= 0) return 0.0f;
  float v_radial = (range_now_km - range_prev_km) / dt_sec;
  return -freq_mhz * 1e6f * v_radial / SPEED_OF_LIGHT;
}

float calc_fspl_db(float range_km, float freq_mhz) {
  if (range_km <= 0) return 0.0f;
  return 32.44f + 20.0f*log10f(freq_mhz) + 20.0f*log10f(range_km);
}

const char* signal_quality(float fspl_db) {
  if (fspl_db < 135.0f) return "EXCELLENT";
  if (fspl_db < 140.0f) return "GOOD";
  if (fspl_db < 145.0f) return "FAIR";
  return "WEAK";
}

const char* signal_bars(float fspl_db) {
  if (fspl_db < 135.0f) return "||||||||";
  if (fspl_db < 138.0f) return "||||||..";
  if (fspl_db < 141.0f) return "||||....";
  if (fspl_db < 144.0f) return "||......";
  return "|.......";
}

void update_telecom_values() {
  if (!device_loc_known || !have_prev_sample) return;
  update_ionospheric_delay();
  float dt = (float)((long)unix_time - (long)prev_unix_time);
  if (dt <= 0.0f) return;
  float iss_range_prev = calc_range_km(device_lat, device_lon, prev_latitude, prev_longitude, ISS_ALT_KM);
  g_iss_range_km  = calc_range_km(device_lat, device_lon, latitude, longitude, ISS_ALT_KM);
  g_iss_elevation = calc_elevation_deg(device_lat, device_lon, latitude, longitude, ISS_ALT_KM);
  g_iss_fspl_db   = calc_fspl_db(g_iss_range_km, ISS_FREQ_MHZ);
  g_iss_doppler_hz = calc_doppler_hz(ISS_FREQ_MHZ, g_iss_range_km, iss_range_prev, dt);
  g_iss_azimuth = calc_azimuth_deg(device_lat, device_lon, latitude, longitude);
  update_pass_state(g_iss_elevation, g_iss_prev_elev, g_iss_pass_state, sizeof(g_iss_pass_state));
  g_iss_prev_elev = g_iss_elevation;
  if (have_css_prev_sample) {
    float css_dt = (float)(unix_time - css_prev_unix_time);
    if(css_dt <= 0) css_dt = dt;
    float css_range_prev = calc_range_km(device_lat, device_lon, css_prev_latitude, css_prev_longitude, CSS_ALT_KM);
    g_css_range_km  = calc_range_km(device_lat, device_lon, css_latitude, css_longitude, CSS_ALT_KM);
    g_css_elevation = calc_elevation_deg(device_lat, device_lon, css_latitude, css_longitude, CSS_ALT_KM);
    g_css_fspl_db   = calc_fspl_db(g_css_range_km, CSS_FREQ_MHZ);
    g_css_doppler_hz = calc_doppler_hz(CSS_FREQ_MHZ, g_css_range_km, css_range_prev, css_dt);
    g_css_azimuth = calc_azimuth_deg(device_lat, device_lon, css_latitude, css_longitude);
    update_pass_state(g_css_elevation, g_css_prev_elev, g_css_pass_state, sizeof(g_css_pass_state));
    g_css_prev_elev = g_css_elevation;
  }
}

void calc_tx_recommendation(float elevation, const char* pass_state, float fspl_db, float kp, const char** text_out, uint32_t* color_out) {
  if (elevation < 0.0f) { *text_out = "NO CONTACT"; *color_out = 0xFF4444; }
  else if (kp >= 6.0f) { *text_out = "STORM - POOR"; *color_out = 0xFF8800; }
  else if (strcmp(pass_state, "SETTING") == 0) { *text_out = "FINISH SOON"; *color_out = 0xFFAA00; }
  else if (strcmp(pass_state, "PEAK") == 0 && fspl_db < 135.0f) { *text_out = "OPTIMAL WINDOW"; *color_out = 0x00FF88; }
  else if (strcmp(pass_state, "RISING") == 0) { *text_out = "TRANSMIT NOW"; *color_out = 0x00FF88; }
  else { *text_out = "IN RANGE"; *color_out = 0x88FF88; }
}

void compute_predictions_for(bool is_iss, PredictedPoint* out) {
  bool t_have = is_iss ? have_prev_sample : have_css_prev_sample;
  if (!t_have) return;
  float t_lat   = is_iss ? latitude          : css_latitude;
  float t_lon   = is_iss ? longitude         : css_longitude;
  float t_plat  = is_iss ? prev_latitude     : css_prev_latitude;
  float t_plon  = is_iss ? prev_longitude    : css_prev_longitude;
  unsigned long t_ptime = is_iss ? prev_unix_time : css_prev_unix_time;
  double lat1 = degrees_to_radians(t_plat), lon1 = degrees_to_radians(t_plon);
  double lat2 = degrees_to_radians(t_lat),  lon2 = degrees_to_radians(t_lon);
  double dt_base = (double)((long)unix_time - (long)t_ptime);
  if (dt_base <= 0) return;
  double dlat = lat2 - lat1, dlon = lon2 - lon1;
  double a = sin(dlat/2)*sin(dlat/2) + cos(lat1)*cos(lat2)*sin(dlon/2)*sin(dlon/2);
  double ang_dist = 2.0 * atan2(sqrt(a), sqrt(1.0 - a));
  double bearing  = atan2(sin(dlon)*cos(lat2), cos(lat1)*sin(lat2) - sin(lat1)*cos(lat2)*cos(dlon));
  double ang_spd  = ang_dist / dt_base;
  const double EARTH_ROT = (2.0 * PI) / 86164.0905;
  for (int i = 0; i < PREDICT_COUNT; i++) {
    double ft   = (double)((i + 1) * PREDICT_STEP_SEC);
    double d    = ang_spd * ft;
    double lat3 = asin(sin(lat2)*cos(d) + cos(lat2)*sin(d)*cos(bearing));
    double lon3 = lon2 + atan2(sin(bearing)*sin(d)*cos(lat2), cos(d) - sin(lat2)*sin(lat3));
    lon3 -= EARTH_ROT * ft;
    double lon3d = lon3 * 180.0 / PI;
    while (lon3d >  180.0) lon3d -= 360.0;
    while (lon3d < -180.0) lon3d += 360.0;
    out[i].lat = (float)(lat3 * 180.0 / PI);
    out[i].lon = (float)lon3d;
    out[i].t   = (unsigned long)unix_time + (unsigned long)ft;
  }
}

// نفس منطق compute_predictions_for بس بترجع نقطة واحدة بدل array
static void sat_pos_at(bool is_iss, unsigned long t_future,
                       float& out_lat, float& out_lon) {
  float  cur_lat = is_iss ? latitude       : css_latitude;
  float  cur_lon = is_iss ? longitude      : css_longitude;
  float  prv_lat = is_iss ? prev_latitude  : css_prev_latitude;
  float  prv_lon = is_iss ? prev_longitude : css_prev_longitude;
  unsigned long prv_t = is_iss ? prev_unix_time : css_prev_unix_time;

  double lat1 = degrees_to_radians(prv_lat), lon1 = degrees_to_radians(prv_lon);
  double lat2 = degrees_to_radians(cur_lat), lon2 = degrees_to_radians(cur_lon);

  double dt0 = (double)((long)unix_time - (long)prv_t);
  if (dt0 < 1.0) { out_lat = cur_lat; out_lon = cur_lon; return; }

  double dlat = lat2 - lat1, dlon = lon2 - lon1;
  double a  = sin(dlat/2)*sin(dlat/2)
            + cos(lat1)*cos(lat2)*sin(dlon/2)*sin(dlon/2);
  double ad = 2.0 * atan2(sqrt(a), sqrt(1.0 - a));
  double br = atan2(sin(dlon)*cos(lat2),
                    cos(lat1)*sin(lat2) - sin(lat1)*cos(lat2)*cos(dlon));
  double spd = ad / dt0;
  double ft  = (double)((long)t_future - (long)unix_time);
  double d   = spd * ft;

  double lat3 = asin(sin(lat2)*cos(d) + cos(lat2)*sin(d)*cos(br));
  double lon3 = lon2 + atan2(sin(br)*sin(d)*cos(lat2),
                              cos(d) - sin(lat2)*sin(lat3));

  // تصحيح دوران الأرض
  lon3 -= (2.0 * PI / 86164.0905) * ft;

  double lon3d = lon3 * 180.0 / PI;
  while (lon3d >  180.0) lon3d -= 360.0;
  while (lon3d < -180.0) lon3d += 360.0;

  out_lat = (float)(lat3 * 180.0 / PI);
  out_lon = (float)lon3d;
}

// يمشي 24 ساعة قدام بخطوات 30 ثانية ويلاقي التمريرات
static int find_passes_local(bool is_iss, float obs_lat, float obs_lon,
                             LocalPass* out, int max_out) {
  const float ALT_KM   = is_iss ? ISS_ALT_KM : CSS_ALT_KM;
  const int   STEP_S   = 30;
  const float MIN_ELEV = 5.0f;       // تجاهل تمريرات أقل من 5 درجة
  const unsigned long WINDOW = 86400UL;

  int   found   = 0;
  bool  in_pass = false;
  float peak_el = -999.0f;
  unsigned long rise_t = 0, peak_t = 0;

  for (unsigned long t = unix_time;
       t < unix_time + WINDOW && found < max_out;
       t += STEP_S) {

    float slat, slon;
    sat_pos_at(is_iss, t, slat, slon);
    float el = calc_elevation_deg(obs_lat, obs_lon, slat, slon, ALT_KM);

    if (!in_pass && el >= MIN_ELEV) {
      in_pass = true; rise_t = t; peak_el = el; peak_t = t;
    } else if (in_pass && el >= MIN_ELEV) {
      if (el > peak_el) { peak_el = el; peak_t = t; }
    } else if (in_pass && el < MIN_ELEV) {
      in_pass = false;
      out[found].rise_utc     = rise_t;
      out[found].peak_utc     = peak_t;
      out[found].max_el_deg   = peak_el;
      out[found].duration_min = max(1, (int)((t - rise_t) / 60));
      found++;
      peak_el = -999.0f;
    }
  }
  return found;
}

static void format_passes_local(const char* sat_name, bool is_iss,
                                float obs_lat, float obs_lon, int& off) {
  off += snprintf(overpass_result + off, sizeof(overpass_result) - off,
                  "-- %s Passes --\n", sat_name);

  LocalPass passes[MAX_PASSES_SHOWN];
  int n = find_passes_local(is_iss, obs_lat, obs_lon, passes, MAX_PASSES_SHOWN);

  if (n == 0) {
    off += snprintf(overpass_result + off, sizeof(overpass_result) - off,
                    "  None visible in 24h\n\n");
    return;
  }
  for (int i = 0; i < n; i++) {
    DateTime dt(passes[i].rise_utc);
    long diff = max(0L, (long)passes[i].rise_utc - (long)unix_time);
    char ts_str[20];
    if (diff < 3600) snprintf(ts_str, sizeof(ts_str), "+%ld min", diff / 60);
    else             snprintf(ts_str, sizeof(ts_str), "+%.1fh",   diff / 3600.0f);

    off += snprintf(overpass_result + off, sizeof(overpass_result) - off,
                    "%d) %s  %02d:%02d UTC\n"
                    "   Max: %.0f deg | %d min\n\n",
                    i + 1, ts_str, dt.hour(), dt.minute(),
                    passes[i].max_el_deg, passes[i].duration_min);
  }
}

// overpassFetchTask — HTTP call واحدة بس (location)، باقيه كله local
// =====================================================================
void overpassFetchTask(void* param) {
  int off = 0;
  overpass_result[0] = '\0';

  // الـ location لازمة مرة واحدة بس وبعدين بتتخزن
  if (!device_loc_known && WiFi.status() == WL_CONNECTED) {
    if (xSemaphoreTake(g_http_mutex, pdMS_TO_TICKS(6000)) == pdTRUE) {
      WiFiClient plain;
      HTTPClient http;
      plain.setTimeout(4000);
      http.setTimeout(4000);
      if (http.begin(plain, "http://ip-api.com/json")) {
        if (http.GET() == 200) {
          DynamicJsonDocument doc(512);
          if (!deserializeJson(doc, http.getString())) {
            device_lat = doc["lat"].as<float>();
            device_lon = doc["lon"].as<float>();
            snprintf(device_city, sizeof(device_city), "%s, %s",
                     doc["city"].as<String>().c_str(),
                     doc["country"].as<String>().c_str());
            device_loc_known = true;
          }
        }
        http.end();
      }
      xSemaphoreGive(g_http_mutex);
    }
  }

  if (g_cancel_overpass) goto end_task;

  if (!device_loc_known) {
    strncpy(overpass_result,
            "Could not get location.\nCheck WiFi & tap time again.",
            sizeof(overpass_result) - 1);
    overpass_fetch_done = true;
    goto end_task;
  }

  if (!have_prev_sample && !have_css_prev_sample) {
    strncpy(overpass_result,
            "Waiting for satellite data.\nTry again in 30 seconds.",
            sizeof(overpass_result) - 1);
    overpass_fetch_done = true;
    goto end_task;
  }

  // كل ده بيحصل على الـ ESP32 نفسه، مش HTTP
  off += snprintf(overpass_result + off, sizeof(overpass_result) - off,
                  "Location: %s\n(%.2f, %.2f)\n\n",
                  device_city, device_lat, device_lon);

  if (have_prev_sample)
    format_passes_local("ISS", true, device_lat, device_lon, off);

  if (g_cancel_overpass) goto end_task;

  if (have_css_prev_sample)
    format_passes_local("Tiangong", false, device_lat, device_lon, off);

  overpass_result[sizeof(overpass_result) - 1] = '\0';
  if (!g_cancel_overpass) overpass_fetch_done = true;

end_task:
  overpassTaskHandle = NULL;
  vTaskDelete(NULL);
}

void loop() {
  lv_timer_handler();
  static uint32_t last_brightness = millis(), last_fact = millis(), last_pred_fade_tick = millis(), last_track_fade_tick = millis();

  if (iss_data_ready) {
    iss_data_ready = false;
    if (unix_time != 0 && staged_unix_time != unix_time) {
      prev_latitude = latitude; prev_longitude = longitude;
      prev_unix_time = unix_time; have_prev_sample = true;
      css_prev_latitude = css_latitude; css_prev_longitude = css_longitude;
      css_prev_unix_time = (staged_css_unix_time != 0) ? staged_css_unix_time : unix_time;
      have_css_prev_sample = true;
    }
    latitude = staged_latitude; longitude = staged_longitude;
    unix_time = staged_unix_time; country_name = String(staged_country_name);
    css_latitude = staged_css_latitude; css_longitude = staged_css_longitude;
    css_country_name = String(staged_css_country_name);
    update_icons_position(); printDataToDisplay(); add_track_points(); update_telecom_values();
    if (!lv_obj_has_flag(telecom_panel, LV_OBJ_FLAG_HIDDEN)) refresh_telecom_panel();
  }

  if (pred_dots_data_ready) { pred_dots_data_ready = false; pred_fade_state = 1; }
  if (millis() - last_pred_fade_tick >= 15) {
    last_pred_fade_tick = millis();
    if (pred_fade_state == 1) {
      bool all_zero = true;
      for (int i = 0; i < PREDICT_COUNT; i++) {
        if (iss_pred_opa[i] > 0) { iss_pred_opa[i] = (uint8_t)max(0, (int)iss_pred_opa[i] - 15); lv_obj_set_style_opa(iss_pred_dots[i], iss_pred_opa[i], 0); if (iss_pred_opa[i] > 0) all_zero = false; }
        if (css_pred_opa[i] > 0) { css_pred_opa[i] = (uint8_t)max(0, (int)css_pred_opa[i] - 15); lv_obj_set_style_opa(css_pred_dots[i], css_pred_opa[i], 0); if (css_pred_opa[i] > 0) all_zero = false; }
      }
      if (all_zero) { apply_pred_dot_positions(); pred_fade_state = 2; }
    } else if (pred_fade_state == 2) {
      bool all_done = true;
      for (int i = 0; i < PREDICT_COUNT; i++) {
        bool show_iss = (g_sat_filter == 0 || g_sat_filter == 1);
        if (show_iss && iss_pred_opa[i] < PRED_SETTLED_OPA[i]) { iss_pred_opa[i] = (uint8_t)min((int)PRED_SETTLED_OPA[i], (int)iss_pred_opa[i] + 15); lv_obj_set_style_opa(iss_pred_dots[i], iss_pred_opa[i], 0); if (iss_pred_opa[i] < PRED_SETTLED_OPA[i]) all_done = false; }
        bool show_css = (g_sat_filter == 0 || g_sat_filter == 2);
        if (show_css && css_pred_opa[i] < PRED_SETTLED_OPA[i]) { css_pred_opa[i] = (uint8_t)min((int)PRED_SETTLED_OPA[i], (int)css_pred_opa[i] + 15); lv_obj_set_style_opa(css_pred_dots[i], css_pred_opa[i], 0); if (css_pred_opa[i] < PRED_SETTLED_OPA[i]) all_done = false; }
      }
      if (all_done) pred_fade_state = 0;
    }
  }

  if (millis() - last_brightness >= BRIGHTNESS_UPDATE_TIME) { if (brightness > SCREEN_MINIMUM_BRIGHTNESS) brightness -= 10; ledcWrite(LCD_BACKLIGHT_PIN, brightness); last_brightness = millis(); }
  if (millis() - last_fact >= FACT_FADE_UPDATE_TIME) { if (fact_opacity > 0) { fact_opacity--; lv_obj_set_style_opa(fact_label, fact_opacity, 0); } last_fact = millis(); }
  if (millis() - last_track_fade_tick >= 500) { update_track_dot_fade(); last_track_fade_tick = millis(); }
  if (overpass_fetch_done) { overpass_fetch_done = false; lv_label_set_text(overpass_label, overpass_result); lv_obj_scroll_to_y(ov_scroll_area, 0, LV_ANIM_OFF); }

  lv_tick_inc(5); delay(5);
}
